+import org.springframework.security.core.userdetails.User;
+import org.springframework.security.core.userdetails.UserDetails;
+import org.springframework.security.core.userdetails.UserDetailsService;
+import org.springframework.security.core.userdetails.UsernameNotFoundException;
+import org.springframework.stereotype.Service;
+
+@Service
+public class AuthzUserDetailsService implements UserDetailsService {
+
+    private final DubboReferenceFactory referenceFactory;
+    private final boolean bootstrapEnabled;
+    private final String bootstrapUsername;
+    private final String bootstrapPassword;
+
+    public AuthzUserDetailsService(
+            DubboReferenceFactory referenceFactory,
+            @Value("${app.bootstrap-admin.enabled:false}") boolean bootstrapEnabled,
+            @Value("${app.bootstrap-admin.username:admin}") String bootstrapUsername,
+            @Value("${app.bootstrap-admin.password:}") String bootstrapPassword) {
+        this.referenceFactory = referenceFactory;
+        this.bootstrapEnabled = bootstrapEnabled;
+        this.bootstrapUsername = bootstrapUsername;
+        this.bootstrapPassword = bootstrapPassword;
+    }
+
+    @Override
+    public UserDetails loadUserByUsername(String username) throws UsernameNotFoundException {
+        AuthzPasswordCredentialDTO credential;
+        try {
+            credential = referenceFactory.get(AuthzUserService.class).getPasswordCredential(username);
+        } catch (Exception ex) {
+            throw new AuthenticationServiceException("认证服务暂不可用", ex);
+        }
+        if (credential != null && !blank(credential.getPasswordHash())) {
+            return User.withUsername(resolvePrincipalName(credential))
+                    .password(credential.getPasswordHash())
+                    .disabled(Boolean.FALSE.equals(credential.getEnabled()))
+                    .authorities(Collections.emptyList())
+                    .build();
+        }
+        if (bootstrapEnabled && bootstrapUsername.equals(username) && !blank(bootstrapPassword)) {
+            return User.withUsername(bootstrapUsername)
+                    .password("{noop}" + bootstrapPassword)
+                    .authorities(Collections.emptyList())
+                    .build();
+        }
+        throw new UsernameNotFoundException("用户不存在或未设置密码");
+    }
+
+    private String resolvePrincipalName(AuthzPasswordCredentialDTO credential) {
+        if (!blank(credential.getUserId())) {
+            return credential.getUserId();
+        }
+        if (!blank(credential.getLoginId())) {
+            return credential.getLoginId();
+        }
+        return credential.getUsername();
+    }
+
+    private boolean blank(String value) {
+        return value == null || value.trim().isEmpty();
+    }
+}
